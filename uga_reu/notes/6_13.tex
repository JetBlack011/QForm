\subsection{Goals}

\begin{enumerate}
	\item Find various examples of $\{\beta,\beta_2\}$ such that the Heegaard diagram $(\Sigma_2, \underbrace{\{\alpha_1,\alpha_2\}}_{\overline{\alpha}}, \underbrace{\{\beta_1,\beta_2\}}_{\overline{\beta}})$ is an integer homology sphere ($\ZHS^3$).
	\item Combine these curves on $\Sigma_2$ to form a trisection diagram, i.e. a diagram $(\Sigma_2, \overline{\alpha}, \overline{\beta}, \overline{\gamma})$ such that $(\overline{\alpha}, \overline{\beta})$, $(\overline{\alpha}, \overline{\gamma})$, and $(\overline{\beta}, \overline{\gamma})$ form Heegaard diagrams for integer homology spheres.
	\item Find cut systems $\overline{\beta},\overline{\gamma},\overline{\delta},\dots$ such that the intersection form given by their genus 2 multisection is of a particular form (e.g. the $E_8$ Cartan matrix).
\end{enumerate}
